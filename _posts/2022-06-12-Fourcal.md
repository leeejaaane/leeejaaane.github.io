```python
class FourCal:
    def setdata(self, first, second):
        self.first = first #객체 변수 : 객체에 생성되는 객체만의 변
        self.second = second

    def add(self):
        result = self.first + self.second
        return result

    def sub(self):
        result = self.first - self.second
        return result

    def mul(self):
        result = self.first * self.second
        return result

    def div(self):
        result = self.first / self.second
        return result

# 객체 선언

a = FourCal()
b = FourCal()

# 객체 변수 세팅

a.setdata(4,2)
print(a.first)
b.setdata(3,7)
print(b.first)

# 사칙연산

a.add() #이미 객체변수로 저장을 해두었기 때문에 메소드 안에 매개 변수 넣어주지 않아도 됨
b.mul()

print(a.add())
print(b.mul())

```

위의 코드는 생성자 함수가 없는 경우이므로 setdata 메소드를 불러와서 객체 변수를 입력해주어야 한다.
생성자 함수를 활용하면 이렇게 코드가 바뀐다.


```python

class FourCal:
    def __init__(self, first, second):
        self.first = first #객체 변수 : 객체에 생성되는 객체만의 변
        self.second = second

    def add(self):
        result = self.first + self.second
        return result

    def sub(self):
        result = self.first - self.second
        return result

    def mul(self):
        result = self.first * self.second
        return result

    def div(self):
        result = self.first / self.second
        return result

# 객체 선언

a = FourCal(4,2)
b = FourCal(3,8)


# 사칙연산

a.add() #이미 객체변수로 저장을 해두었기 때문에 메소드 안에 매개 변수 넣어주지 않아도 됨
b.mul()

print(a.add())
print(b.mul())
```

거북이로 글자 그래픽 만들기

```python
import turtle
import random

def getXY():
    x,y,angle,size= 0,0,0,0
    x = random.randint(-250, 250)
    y = random.randint(-250,250)
    size = random.randint(10, 50)
    angle = random.randint(0,360)
    return x,y,angle,size

korean = """가나다라마바"""

turtle.penup()
turtle.speed(5)

for ch in korean:
    x,y,angle,size = getXY() #함수에서 리턴된 값 받아오기
    turtle.goto(x,y)
    turtle.write(ch, font=("맑은고딕", size, 'bold'))
turtle.done
```

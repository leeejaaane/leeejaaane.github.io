class FourCal:
    def setdata(self, first, second):
        self.first = first #객체 변수 : 객체에 생성되는 객체만의 변
        self.second = second

    def add(self):
        result = self.first + self.second
        return result

    def sub(self):
        result = self.first - self.second
        return result

    def mul(self):
        result = self.first * self.second
        return result

    def div(self):
        result = self.first / self.second
        return result

# 객체 선언

a = FourCal()
b = FourCal()

# 객체 변수 세팅

a.setdata(4,2)
print(a.first)
b.setdata(3,7)
print(b.first)

# 사칙연산

a.add() #이미 객체변수로 저장을 해두었기 때문에 메소드 안에 매개 변수 넣어주지 않아도 됨
b.mul()

print(a.add())
print(b.mul())

## Python 리스트와 append

​    #리스트 선언
​        numList = []

​    #for문으로 리스트에 값 대입

​    for i in range(0,4):
​            numList.append(0)
​            print(i,"번째 값:",numList[i])

​    hap = 0

​    for i in range(0,4) :
​            numList[i] = int(input("숫자:"))

​    hap = numList[0] + numList[1] + numList[2] + numList[3]

​    print("합계 ==>", hap)

## 리스트의 연산

> numList = [10, 20, 30]
> 
> myList = [40, 50, 60]
> 
> print(numList + myList)
> 
> print(numList * 3)
> 

> #연속된 범위의 값 변경
> 
> numList = [10, 20, 30]
> 

> numList[2:3] = [200, 201]
>
> print(numList)
>
> #만약 하나의 첨자만 사용해서 값을 변경하는 경우
>
> numList = [10, 20, 30]
> 
> numList[2] = [200, 201]
>
> print(numList)
>
> #insert()로 정해진 위치에 값 삽입하기
>
> numList = [10, 20, 30]
> 
> numList.insert(1, 300)
> 
> print(numList)


# 재고 확인 프로그램

> store = {}
> 
> while True:
> 
>     name = input('물품명:')
> 
>     if name == "z":
> 
>         break
>     
>     num = input('재고:')
> 
>     store[name] = [num]
> 
> 
> #물품명 입력 받기
> 
> while True:
> 
>     item = input("찾을 물품 =>")
> 
>     if item == "":
> 
>         break
> 
>     if item in store:
> 
>         print(store[item], "개 남았어요")
> 
>     else:
> 
>         print("그 물품은 없어요")

##[Python]리스트와 append

        #리스트 선언
        numList = []

        #for문으로 리스트에 값 대입

        for i in range(0,4):
            numList.append(0)
            print(i,"번째 값:",numList[i])

        hap = 0

        for i in range(0,4) :
            numList[i] = int(input("숫자:"))

        hap = numList[0] + numList[1] + numList[2] + numList[3]

        print("합계 ==>", hap)
